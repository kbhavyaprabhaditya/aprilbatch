    --reuse-values --set green.enabled=true --set green.tag=<new-tag>

  # Switch traffic to green:
  helm upgrade {{ .Release.Name }} ./helm/java-web-app -n {{ .Release.Namespace }} \
    --reuse-values --set activeSlot=green

  # Cleanup old blue slot:
  helm upgrade {{ .Release.Name }} ./helm/java-web-app -n {{ .Release.Namespace }} \
    --reuse-values --set blue.enabled=false
